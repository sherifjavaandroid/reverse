استخراج وفك APK

فك ضغط APK مباشرة

BASIC 
# طريقة 1: unzip مباشر (بدون فك تشفير الـ manifest)

# طريقة 2: فك بـ 7z (يدعم ملفات أكبر)
7z x "miqat user.apk" -omiqat_extracted

فك APK بـ apktool (يفك تشفير XML + resources)

RECOMMENDED 
# فك كامل مع فك تشفير الـ resources و manifest
apktool d "miqat user.apk" -o miqat_decoded

# فك بدون فك resources (أسرع - كافي لـ Flutter)
apktool d "miqat user.apk" -o miqat_decoded -r

# فك بدون فك الكود (smali)
apktool d "miqat user.apk" -o miqat_decoded -s

# فك مع force overwrite
apktool d "miqat user.apk" -o miqat_decoded -f

Hook على دالة Dart محددة (من Blutter output)

DART 
// hook_dart_function.js
// بعد تشغيل Blutter، ابحث عن عنوان الدالة من asm/ files
// مثال: grep -rn "login" miqat_output/asm/
// Windows: findstr /s /i /n "login" miqat_output\asm\*.txt

var libapp = Module.findBaseAddress('libapp.so');

if (libapp) {
    console.log('[+] libapp.so at: ' + libapp);

    // عنوان الدالة من Blutter output (asm/ files)
    // استبدل 0xXXXXXX بالعنوان الحقيقي
    var loginFuncAddr = libapp.add(0xXXXXXX);

    Interceptor.attach(loginFuncAddr, {
        onEnter: function(args) {
            console.log('[+] Login function called!');
            console.log('    arg0: ' + args[0]);
            console.log('    arg1: ' + args[1]);
        },
        onLeave: function(retval) {
            console.log('[+] Login returned: ' + retval);
        }
    });
} else {
    console.log('[-] Waiting for libapp.so...');
    // إذا التطبيق لسا ما حمّل libapp
    var interval = setInterval(function() {
        libapp = Module.findBaseAddress('libapp.so');
        if (libapp) {
            clearInterval(interval);
            console.log('[+] libapp.so loaded at: ' + libapp);
        }
    }, 500);
}

Hook على HTTP requests (اعتراض كل الطلبات)

NETWORK 
// hook_http.js - اعتراض طلبات HTTP عبر Java layer

Java.perform(function() {
    // Hook OkHttp3
    var OkHttpClient = Java.use('okhttp3.OkHttpClient');
    var Request = Java.use('okhttp3.Request');

    var RealCall = Java.use('okhttp3.internal.connection.RealCall');
    RealCall.execute.implementation = function() {
        var request = this.request();
        console.log('[HTTP] ' + request.method() + ' ' + request.url().toString());

        // عرض Headers
        var headers = request.headers();
        for (var i = 0; i < headers.size(); i++) {
            console.log('    ' + headers.name(i) + ': ' + headers.value(i));
        }

        var response = this.execute();
        console.log('[RESP] ' + response.code());
        return response;
    };

    // Hook URL connections
    var URL = Java.use('java.net.URL');
